Step 1: Get API keys from user

      User's API key is located in Options > Account Settings
      Organization's guest user API key is located in Options > Admin > Integrations

myApp.controller('getUser', ['$scope', '$http', function($scope, $http){

$scope.showUser = false;

$http.get('/users/username').success(function(data){

console.log("Username response: ", data);

if (data == "false"){

$scope.showLogin = true;

} else {

$scope.showUser = true;

$scope.username = data.username;

}

});

$scope.logout = function() {

$http.get('/users/logout').success(function(data){

$scope.showLogin = true;

$scope.showUser = false;

console.log(data + " has been logged out");

})

}

}]);
